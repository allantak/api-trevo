 import javax.validation.constraints.Email;

 public class ClientForm {

     private String clientName;
 
     @NotBlank

     private String email;
 
     @NotBlank(message = "Obrigatório o telefone")
     private String phone;
 
     public String getClientName() {
         return clientName;
     }

         return email;
     }

     }

     }

         this.phone = phone;
     }
 }
